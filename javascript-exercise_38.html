מערכים תרגיל מספר 38 מציאת DNA

בתרגיל ג'אווה סקריפט הבא נלמד איך לייצר מחרוזת ספציפית מתוך מערך קיים, התרגיל הינו לבחור אות אחת מתוך מערך קיים כל פעם עד שנקבל שלושה אותיות בסדר ספציפי תוך כדי שמירה על ביצועי מהירים וחישוב מדויק

  • a. create a variable myChain,that holds an empty string ''.
  • b. create a variable charBank that holds this array ["r","n","a","d","b","p"]
  • c. create a variable numOfIterations that holds 0 as initial state.
  • d. create a variable safeGuard that holds 1000 as initial state.
  • e.write a loop that randomly picks 3 chars from the charBank, join it into myChain as a string.
  • f. print how many iterations you have to run until myChain becomes DNA or RNA.
  • g. Use safeGuard to succeed this with a while loop so your computer won't crash a lot :) When you finish and everything works, remove it . 

const charBank = ['r', 'n', 'a', 'd', 'b', 'p']
let myChain = ''
let numOfIterations = 0
let safeGuard = 1000

while (myChain !== 'rna' || myChain !== 'dna' || numOfIterations < safeGuard) {
    numOfIterations++
    for (let i = 0; i < 3; i++) {
        const randomIdx = getRandomIntInclusive(0, charBank.length - 1)
        myChain += charBank[randomIdx]
    }
    if (myChain === 'dna' || myChain === 'rna') {
        break
    }
    myChain = ''
}

const getRandomIntInclusive = (min, max) =>
    Math.floor(
        Math.random() * (Math.floor(max) - Math.ceil(min) + 1) + Math.ceil(min)
    )

console.log('it took', numOfIterations, 'iterations to find', myChain)
