בתרגיל ג'אווה סקריפט הבא נלמד איך למלא מערך ריק במספרים

  • a. prompt the user to enter how many elements he wants to create in a new array save as numOfElements.
  • b. write a function fillArray() that generate new array, fill it with random numbers, with the numOfElements the user request.
  • c. Bonus: make sure only ODD numbers can enter the new array. 

const numOfElements = +prompt('What is the size of the array you want?')

const getOddNumber = () => {
    let odd
    while (!odd) {
        const randomNum = getRandomIntInclusive(0, 100)
        if (randomNum % 2) {
            odd = randomNum
        }
    }
    return odd
}

const getRandomIntInclusive = (min, max) =>
    Math.floor(
        Math.random() * (Math.floor(max) - Math.ceil(min) + 1) + Math.ceil(min)
    )

const fillArray = () => {
    let filledArray = []
    for (let i = 0; i < numOfElements; i++) {
        const randomOddNumber = getOddNumber(0, 100)
        filledArray.push(randomOddNumber)
    }
    return filledArray
}

const filledArray = fillArray()
console.log('filledArray', filledArray)

/**
 * Short way not always easier to understand!
 */
const filledArray2 = [...Array(numOfElements)].map(() => getOddNumber(0, 100))
console.log('filledArray2', filledArray2)
