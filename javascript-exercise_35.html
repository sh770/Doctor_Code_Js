מערכים תרגיל מספר 35 איחוד מערכים

בתרגיל ג'אווה סקריפט הבא נלמד איך לאחד שתי מערכים שונים לידי מערך אחד גדול שכולל בתוכו מופעים בודדים של איברים שחוזרים על עצמם

Array a ['🧨', '🧨', '🧨', '🧨']

Array b ['🍌', '🍌', '🍌', '🧨', '🧨', '💧']

Expected result => [🧨,🍌,💧]

  • a. write a function mergeArrays() to merge two arrays and removes all duplicates elements.
  • b. first use for or while loops then learn to use array.filter() or array.concat(). 

const array1 = ['🧨', '🧨', '🧨', '🧨']
const array2 = ['🍌', '🍌', '🍌', '🧨', '🧨', '💧']

// the hard way
const mergeArrays = (arrayNum1, arrayNum2) => {
    let arrayNum3 = []
    arrayNum1.forEach((item) => {
        arrayNum3.push(item)
    })
    arrayNum2.forEach((item) => {
        arrayNum3.push(item)
    })
    return arrayNum3.filter(
        (value, index) => arrayNum3.indexOf(value) === index
    )
}
// the hard way
const array3 = mergeArrays(array1, array2)
console.log('merged cleaned array: array3', array3)

const filterByIndex = (v, i, s) => s.indexOf(v) === i

// easier way
const array4 = [...array1, ...array2].filter(filterByIndex)
console.log('using es6 power => ', array4)

// easier way 2
const array5 = array1.concat(array2).filter(filterByIndex)
console.log('using es6 concat', array5)
