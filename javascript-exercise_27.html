בתרגיל ג'אווה סקריפט הבא נלמד איך לקבל לרשום פונקציה שמקבלת מערך של מספרים ודואגת לנקות את המערך ממספרים ששווים לאפס ולהחזיר מערך חדש לאחר מכן

  • a. create a function formatArray() that gets an numArray and excludeNum parameter and returns a new array clean of this excludeNum.
  • b. first use for or while loops then learn to use array.filter().
  • c. if we want to exclude "0" [0, 1, 0, 5, 6, 7, 9, 4, 2, 0, 0, 0, 1] will become [1, 5, 6, 7, 9, 4, 2, 1] //clean of zeroes. 

let numArray = [0, 1, 0, 5, 6, 7, 9, 4, 2, 0, 0, 0, 1]

/**
 * Level 1 - 8 lines
 */

const formatArray = (numArray, excludeNum) => {
    const result = []
    for (let i = 0; i < numArray.length; i++) {
        if (numArray[i] !== excludeNum) {
            result.push(numArray[i])
        }
    }
    return result
}

console.log(formatArray(numArray, 0))

/**
 * Level 2 - 4 lines
 */

const formatArrayTwo = (numArray, excludeNum) => {
    return numArray.filter((item) => {
        return item !== excludeNum
    })
}
console.log(formatArrayTwo(numArray, 0))

/**
 * Level 3 - one line
 * a - for array
 * e - for exclude
 * i - for iteration item
 */
const formatArrayThree = (a, e) => a.filter((i) => i !== e)
console.log(formatArrayThree(numArray, 0))
