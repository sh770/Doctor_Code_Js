בתרגיל ג'אווה סקריפט הבא נלמד איך לקבל מערך של מחרוזות לחבר את המחרוזות במערך לכדי מחרוזת אחת חדשה שכוללת את כל איברי המערך

  • a. consider this array ['let', 'the', 'games', 'begin'].
  • b. create a function joinArray() which gets array of words and return a sentence as string with spaces, 'let the games begin'
  • b. first use for or while loops then learn to use array.join(). 

let strArray = ['let', 'the', 'games', 'begin']
const joinArray = (strArray) => {
    let strSeq = ''
    for (let i = 0; i < strArray.length; i++) {
        if (i === 0 ){
            strSeq += strArray[i]
        } else {
            strSeq = strSeq + ' ' + strArray[i]
        }
    }
    return strSeq
}
alert(joinArray(strArray))

/**
 * joinArrayTwo- one line using array.join()
 */
const joinArrayTwo = (strArray) => strArray.join('_')
alert(joinArrayTwo(strArray))
