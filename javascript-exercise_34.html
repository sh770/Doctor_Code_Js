מערכים תרגיל מספר 34 הסרת כפילויות

בתרגיל ג'אווה סקריפט הבא נלמד איך לקבל מהמשתמש מערך של משתנים ובאמצעות לולאה פשוטה איך להסיר כפילויות במערך

  • a. using the following array: [“jsJP6”,”gBAqd”,”y2TC1”,”jsJP6”,”y2TC1”]
  • b. write a function that gets rid of duplicates, returns a new array and prints it. 

const idList = ['jsJP6', 'gBAqd', 'y2TC1', 'jsJP6', 'y2TC1']

const removeArrayDuplicates = (arr) => {
    let uniqueArray = []
    arr.forEach((element) => {
        if (uniqueArray.indexOf(element) === -1) {
            uniqueArray.push(element)
        }
    })
    return uniqueArray
}

const uniqueIds = removeArrayDuplicates(idList)
console.log('Clean Array', uniqueIds)
