בתרגיל ג'אווה סקריפט הבא נלמד איך להשתמש בלולאה כדי לעבור על מערך ולהוציא ממנו כל ערך שאינו ערך אמת

  • a. consider this array [null, 0, 0, 1, 7, '', 'banana'].
  • b. write a function cleanFalsyArray() that gets the array and returns a new array with only values that are truthy and print it. 

let customArray = [null, 0, 0, 1, 7, '', 'banana']
const cleanFalsyArray = (arr) => {
    let cleanArr = []
    for (let i = 0; i < arr.length; i++) {
        if (arr[i]) {
            cleanArr.push(arr[i])
        }
    }
    return cleanArr

}
const cleanedArray = cleanFalsyArray(customArray)
console.log('cleaned array', cleanedArray)
